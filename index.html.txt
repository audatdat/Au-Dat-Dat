  <!-- Nút về đầu trang -->
  <button id="backToTop" class="back-to-top" aria-label="Về đầu trang">↑</button>

  <script>
    // Chỉ chạy khi toàn bộ DOM đã sẵn sàng
    document.addEventListener('DOMContentLoaded', function () {
      const mobileMenuBtn = document.getElementById('mobileMenuBtn');
      const mobileMenu = document.getElementById('mobileMenu');
      const backToTop = document.getElementById('backToTop');
      const currentYear = document.getElementById('currentYear');
      const revealItems = document.querySelectorAll('.reveal');
      const timelineItems = document.querySelectorAll('.timeline-item');
      const skillCards = document.querySelectorAll('.skill-card');
      const skillFills = document.querySelectorAll('.fill');
      const counterNodes = document.querySelectorAll('[data-counter]');
      const navLinks = document.querySelectorAll('#mobileMenu a');
      const heroAvatar = document.getElementById('heroAvatar');
      const avatarPhoto = document.getElementById('avatarPhoto');
      const avatarUpload = document.getElementById('avatarUpload');
      const avatarStorageKey = 'au-phat-dat-avatar';

      if (currentYear) {
        currentYear.textContent = new Date().getFullYear();
      }

      // Ưu tiên hiển thị ảnh thật làm avatar: file profile-photo.jpg hoặc ảnh người dùng chọn trực tiếp từ máy
      function setAvatarFallbackState(isFallback) {
        if (!heroAvatar) return;
        heroAvatar.classList.toggle('avatar-fallback', isFallback);
      }

      if (avatarPhoto) {
        avatarPhoto.addEventListener('load', function () {
          setAvatarFallbackState(false);
        });

        avatarPhoto.addEventListener('error', function () {
          setAvatarFallbackState(true);
        });
      }

      try {
        const storedAvatar = window.localStorage.getItem(avatarStorageKey);
        if (storedAvatar && avatarPhoto) {
          avatarPhoto.src = storedAvatar;
        }
      } catch (error) {
        // Bỏ qua nếu trình duyệt chặn localStorage
      }

      if (heroAvatar && avatarUpload && avatarPhoto) {
        const openAvatarPicker = function () {
          avatarUpload.click();
        };

        heroAvatar.addEventListener('click', openAvatarPicker);
        heroAvatar.addEventListener('keydown', function (event) {
          if (event.key === 'Enter' || event.key === ' ') {
            event.preventDefault();
            openAvatarPicker();
          }
        });

        avatarUpload.addEventListener('change', function () {
          const selectedFile = this.files && this.files[0];
          if (!selectedFile || !selectedFile.type.startsWith('image/')) return;

          const reader = new FileReader();
          reader.onload = function (event) {
            const result = typeof event.target?.result === 'string' ? event.target.result : '';
            if (!result) return;

            avatarPhoto.src = result;
            setAvatarFallbackState(false);

            try {
              window.localStorage.setItem(avatarStorageKey, result);
            } catch (error) {
              // Bỏ qua nếu ảnh quá lớn hoặc localStorage không khả dụng
            }
          };

          reader.readAsDataURL(selectedFile);
        });
      }

      // Điều khiển menu mobile
      if (mobileMenuBtn && mobileMenu) {
        mobileMenuBtn.addEventListener('click', function () {
          const isOpen = mobileMenu.classList.toggle('open');
          mobileMenuBtn.classList.toggle('active', isOpen);
          mobileMenuBtn.setAttribute('aria-expanded', String(isOpen));
        });

        navLinks.forEach(function (link) {
          link.addEventListener('click', function () {
            mobileMenu.classList.remove('open');
            mobileMenuBtn.classList.remove('active');
            mobileMenuBtn.setAttribute('aria-expanded', 'false');
          });
        });
      }

      // Hiển thị nút về đầu trang khi cuộn xuống
      function toggleBackToTop() {
        if (window.scrollY > 480) {
          backToTop.classList.add('show');
        } else {
          backToTop.classList.remove('show');
        }
      }

      window.addEventListener('scroll', toggleBackToTop, { passive: true });
      toggleBackToTop();

      if (backToTop) {
        backToTop.addEventListener('click', function () {
          window.scrollTo({ top: 0, behavior: 'smooth' });
        });
      }

      // Intersection Observer cho các phần tử fade-in
      const revealObserver = new IntersectionObserver(
        function (entries, observer) {
          entries.forEach(function (entry) {
            if (entry.isIntersecting) {
              entry.target.classList.add('visible');
              observer.unobserve(entry.target);
            }
          });
        },
        { threshold: 0.16 }
      );

      revealItems.forEach(function (item) {
        revealObserver.observe(item);
      });

      timelineItems.forEach(function (item) {
        revealObserver.observe(item);
      });

      // Animation cho thanh kỹ năng và hiệu ứng "nảy mầm"
      let skillsAnimated = false;
      const skillSection = document.getElementById('ky-nang');

      function animateSkills() {
        if (skillsAnimated) return;
        skillsAnimated = true;

        skillFills.forEach(function (fill) {
          const targetWidth = fill.getAttribute('data-width') || '0%';
          fill.style.width = targetWidth;
        });

        skillCards.forEach(function (card, index) {
          window.setTimeout(function () {
            card.classList.add('grown');
          }, index * 100);
        });
      }

      if (skillSection) {
        const skillObserver = new IntersectionObserver(
          function (entries, observer) {
            entries.forEach(function (entry) {
              if (entry.isIntersecting) {
                animateSkills();
                observer.unobserve(entry.target);
              }
            });
          },
          { threshold: 0.22 }
        );

        skillObserver.observe(skillSection);
      }

      // Bộ đếm số liệu ở hero
      let counterPlayed = false;
      const counterSection = document.getElementById('counterSection');

      function runCounters() {
        if (counterPlayed) return;
        counterPlayed = true;

        counterNodes.forEach(function (node) {
          const target = Number(node.getAttribute('data-counter')) || 0;
          const duration = 1400;
          const startTime = performance.now();

          function updateCounter(now) {
            const progress = Math.min((now - startTime) / duration, 1);
            const eased = 1 - Math.pow(1 - progress, 3);
            const value = Math.floor(eased * target);
            node.textContent = String(value);

            if (progress < 1) {
              requestAnimationFrame(updateCounter);
            } else {
              node.textContent = String(target);
            }
          }

          requestAnimationFrame(updateCounter);
        });
      }

      if (counterSection) {
        const counterObserver = new IntersectionObserver(
          function (entries, observer) {
            entries.forEach(function (entry) {
              if (entry.isIntersecting) {
                runCounters();
                observer.unobserve(entry.target);
              }
            });
          },
          { threshold: 0.4 }
        );

        counterObserver.observe(counterSection);
      }
    });
  </script>
